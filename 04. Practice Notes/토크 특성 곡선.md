### 코드

```python
import numpy as np
import matplotlib.pyplot as plt

# 부하 전류 범위 설정 (0~15A로 확장해 감소 구간 관찰)
I = np.linspace(0, 15, 150)

# 상수 값 (현실적 특성 반영)
K = 0.1    # 전동기 상수
kf = 0.05  # 직권 계자 상수
If = 2.0   # 분권 계자 전류 (일정)
k_shunt = 0.05  # 분권 계자 상수
k_series = 0.04 # 직권 계자 상수 (복권에서 사용)
k_sat = 0.2     # 포화 상수 (직권 자속 포화 반영)

# 각 전동기의 토크 계산
# 직권: T = K * kf * I^2 (포화 고려)
phi_series = kf * I / (1 + k_sat * I)  # 포화 효과 추가
T_series = K * phi_series * I

# 분권: T = K * k_shunt * If * I
T_shunt = K * k_shunt * If * I

# 가동 복권: T = K * (k_shunt * If + k_series * I / (1 + k_sat * I)) * I
phi_cumulative = k_shunt * If + k_series * I / (1 + k_sat * I)  # 직권 자속 포화
T_cumulative = K * phi_cumulative * I

# 차동 복권: T = K * (k_shunt * If - k_series * I / (1 + k_sat * I)) * I
phi_differential = k_shunt * If - k_series * I / (1 + k_sat * I)  # 직권 자속 포화
# 총 자속이 0 이하로 떨어지지 않도록 최소값 제한, 현실적 감소 반영
phi_differential = np.maximum(phi_differential, 0.01)  # 자속이 완전히 0이 되지 않음
T_differential = K * phi_differential * I

# 그래프 그리기
plt.figure(figsize=(12, 7))
plt.plot(I, T_series, label='직권 (Series)', color='blue')
plt.plot(I, T_shunt, label='분권 (Shunt)', color='green')
plt.plot(I, T_cumulative, label='가동 복권 (Cumulative Compound)', color='red')
plt.plot(I, T_differential, label='차동 복권 (Differential Compound)', color='purple')

# 그래프 설정
plt.title('부하 전류에 따른 전동기 토크 비교')
plt.xlabel('부하 전류 (A)')
plt.ylabel('토크 (N·m)')
plt.grid(True)
plt.legend()
plt.xlim(0, 4)  # X축 범위 0~4로 제한
plt.ylim(0, max(T_series[40], T_cumulative[40]) * 1.1)  # Y축 범위 동적 조정 (0~4 범위 토크 기반)

# 그래프 출력
plt.show()
```
